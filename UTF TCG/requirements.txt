# UTF Tactics TCG — Requirements
# Node.js project — this file documents runtime and deploy requirements.
# Actual JS dependencies are in package.json.

# ── Runtime ───────────────────────────────────────────────────────────────────
node >= 18.0.0
npm  >= 9.0.0

# ── Node dependencies (see package.json) ─────────────────────────────────────
express@^4.18.2      # HTTP server — serves tcg.html as a static file

# ── No other dependencies ─────────────────────────────────────────────────────
# tcg.html is a single self-contained file:
#   - No React, Vue, Angular
#   - No bundler (webpack, vite, etc.)
#   - No TypeScript
#   - No external CSS frameworks
#   - No CDN-loaded scripts
#   - Canvas 2D API (built into browser)
#   - requestAnimationFrame (built into browser)

# ── Browser support ───────────────────────────────────────────────────────────
# Chrome  >= 90
# Firefox >= 88
# Safari  >= 14
# Edge    >= 90
# Mobile Chrome/Safari supported (mobile-first layout, max-width 430px)
# NOTE: Canvas animations require http:// or https:// — do NOT open via file://

# ── Deploy (Render.com) ───────────────────────────────────────────────────────
# Build command:  npm install
# Start command:  node server.js
# Environment:    Node
# Port:           process.env.PORT (Render injects this automatically)
